Argumenty funkcji javascript

W języku javascript przekazywanie argumentów do funkcji może odbywać się w sposób tradycyjny, czyli poprzez określoną ilość zadeklarowanych wcześniej argumentów:

function zwrocArgumentyFunkcji(arg0, arg1, arg2) { return arg0 + " " + arg1 + " " + arg2; };
zwrocArgumentyFunkcji(1, "Kot", 1.17) >> "1 Kot 1.17"

Innym sposobem przekazywania i dostępu do argumentów funkcji w języku javascript jest przekazywanie dowolnej ilości argumentów i odwoływanie się wewnątrz funkcji do nich poprzez obiekt arguments:

function zwrocArgumentyFunkcji() { return arguments; } function zwrocIloscArgumentowFunkcji() { return arguments.length; } function wypiszArgumentyFunkcji() { for(var i = 0; i < arguments.length; i++) { console.log(arguments[i]); } }
zwrocArgumentyFunkcji(1, "Kot", 1.17) >> Arguments { 0: 1, 1: "Kot", 2: 1.17, … } zwrocIloscArgumentowFunkcji(1, "Kot", 1.17) >> 3 wypiszArgumentyFunkcji(1, "Kot", 1.17) >> 1 >> Kot >> 1.17
