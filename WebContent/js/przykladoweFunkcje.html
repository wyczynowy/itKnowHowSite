Przykładowe funkcje

Poniżej przykłady użycia funkcji w celu uzyskania określonego działania:

Animowanie elementu (widoczność)

var przelacznik = 'zmniejsz'; var element = document.getElementsByTagName('img')[0]; if(element.style.opacity!=1.0)element.style.opacity = 1.0; // Zabezpieczenie przed brakiem określonego parametru opacity function animujOpacity() { var opacity = parseFloat(element.style.opacity); if(przelacznik=='zwiększ') { element.style.opacity = opacity+0.1; if(element.style.opacity>=1.0) przelacznik = 'zmniejsz'; } else { element.style.opacity = opacity-0.1; if(element.style.opacity<=0.0) przelacznik = 'zwiększ'; } } var id = setInterval(animujOpacity,100) // Uruchomienie animacji clearInterval(id) // Zatrzymanie animacji

Funkcja sleep()

function sleep(milliseconds) { var start = new Date().getTime(); for (var i = 0; i < 1e7; i++) { if ((new Date().getTime() - start) > milliseconds){ break; } } }

Załadowanie jquery po załadowaniu strony

var script = document.createElement('script'); script.src = 'https://code.jquery.com/jquery-1.11.0.min.js'; script.type = 'text/javascript'; document.getElementsByTagName('head')[0].appendChild(script);

Uruchomienie funkcji po załadowaniu strony

if (window.addEventListener) window.addEventListener("load", funkcja, false); else if (window.attachEvent) window.attachEvent("onload", funkcja); else window.onload = funkcja;
