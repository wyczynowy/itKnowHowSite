Operacje na plikach

Dostęp do plików w folderze w którym znajduje się program:

package app; import java.io.File; public class Main { public static void main(String[] args) { File folder = new File("./"); File[] listOfFiles = folder.listFiles(); for (File file : listOfFiles) { System.out.println("Nazwa pliku: " + file.getName()); System.out.println("Ścieżka absolutna: " + file.getAbsolutePath()); } } }

Odczyt danych z pliku:

package app; import java.io.BufferedReader; import java.io.FileReader; import java.io.IOException; public class Main { public static void main(String[] args) { BufferedReader bufferedReader = null; try { bufferedReader = new BufferedReader(new FileReader("C:\\exampleFolder\\exampleFile.txt")); String readedLine = ""; while ((readedLine = bufferedReader.readLine()) != null) { System.out.println(readedLine); } } catch (Exception e) { e.printStackTrace(); } finally { if (bufferedReader != null) { try { bufferedReader.close(); } catch (IOException e) { e.printStackTrace(); } } } } }

Zapis danych do pliku:

package app; import java.io.BufferedWriter; import java.io.FileWriter; import java.io.IOException; public class Main { public static void main(String[] args) { BufferedWriter bufferedWriter = null; try { bufferedWriter = new BufferedWriter(new FileWriter("C:\\exampleFolder\\exampleFile.txt")); bufferedWriter.write("Linia pierwsza \r\n"); bufferedWriter.write("Linia druga \r\n"); bufferedWriter.write("Linia trzecia \r\n"); } catch (Exception e) { e.printStackTrace(); } finally { if (bufferedWriter != null) { try { bufferedWriter.close(); } catch (IOException e) { e.printStackTrace(); } } } } }
