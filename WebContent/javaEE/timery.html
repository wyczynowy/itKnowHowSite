Timery

Timery w Javie to nic innego jak działające wątki, wykonujące określone zadania w ramach tych wątków. Przykład poniżej:

package app; import java.util.Date; import java.util.Timer; import java.util.TimerTask; // Klasa rozszerzająca TimerTask to klasa "zadanie". Zadanie to będzie wykonywane w ramach wątku timera class TimerExample extends TimerTask { private String name; public TimerExample(String name) { super(); this.name = name; } @Override public void run() { System.out.println("Licznik: " + Thread.currentThread().getName() + ", " + name + " wykonano z sukcesem - " + new Date()); } } public class Main { public static void main(String[] args) { // Tworzymy timer i kolejno dodajemy zadania jakie ma wykonywać Timer timer = new Timer("Licznik1"); timer.scheduleAtFixedRate(new TimerExample("Zadanie1"), 0, 5 * 1000); timer.scheduleAtFixedRate(new TimerExample("Zadanie2"), 0, 1 * 1000); } }
